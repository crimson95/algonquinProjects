SELECT 
    g.GAME_ID,
    g.NAME AS GAME_NAME,
    p.NAME AS PUBLISHER_NAME,
    gen.GENRE_NAME AS GAME_GENRE,
    v.VERSION_STRING,
    gv.START_DATE,
    gv.END_DATE,
    gv.NOTES
FROM 
    GAME g
JOIN 
    GAME_VERSION gv ON g.GAME_ID = gv.GAME_ID
JOIN 
    VERSION v ON gv.VERSION_ID = v.VERSION_ID
LEFT JOIN 
    PUBLISHER p ON g.PUBLISHER_ID = p.PUBLISHER_ID
LEFT JOIN 
    GENRE gen ON g.GENRE_ID = gen.GENRE_ID
ORDER BY 
    g.GAME_ID, gv.START_DATE;
